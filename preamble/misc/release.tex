\newcommand*{\ifgitinfo}{%
	\ifdefined\gitVtags%
		\sffamily%
		\footnotesize%
		rev\gitVtags:\href{https://github.com/OpenLogicProject/OpenLogic/commits/master}{\gitAbbrevHash}%
		(\gitAuthorDate)%
	\else%
		\relax%
	\fi%
}
